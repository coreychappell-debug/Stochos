<script>
(function(){
  'use strict';

  const state = {
    fiscalYear: 2026,
    totalSalesTarget: 0,
    denominations: [],
    games: [],
    vendorPricing: {},
    featureMasterList: []
  };

  // --- Utilities ------------------------------------------------------------
  function logAudit(msg){
    const wrap = document.getElementById('audit');
    if(!wrap) return;
    const when = new Date().toLocaleString();
    const el = document.createElement('div');
    el.className = 'item';
    el.textContent = `${when} — ${msg}`;
    wrap.prepend(el);
  }

  // Robust CSV parser (handles quoted fields and commas/newlines inside quotes)
  function parseCSV(text){
    const rows = [];
    let i=0, field='', row=[], inQuotes=false;
    const pushField=()=>{ row.push(field); field=''; };
    const pushRow=()=>{ rows.push(row); row=[]; };
    while(i<text.length){
      const c=text[i];
      if(inQuotes){
        if(c==='"'){
          if(text[i+1]==='"'){ field+='"'; i++; } else { inQuotes=false; }
        } else { field+=c; }
      } else {
        if(c==='"'){ inQuotes=true; }
        else if(c===','){ pushField(); }
        else if(c==='
'){ pushField(); pushRow(); }
        else if(c==='
'){ /* ignore */ }
        else { field+=c; }
      }
      i++;
    }
    if(field!==''||row.length){ pushField(); pushRow(); }
    const headers = rows.shift().map(h=>h.trim());
    return rows.filter(r=>r.length&&r.some(v=>v!==''))
               .map(r=>Object.fromEntries(headers.map((h,idx)=>[h, (r[idx]??'').trim()])));
  }

  function toCSV(arr){
    if(!arr.length) return '';
    const headers = Object.keys(arr[0]);
    const esc=v=>{
      if(v==null) return '';
      const s=String(v);
      return /[",
]/.test(s) ? '"'+s.replace(/"/g,'""')+'"' : s;
    };
    const lines=[headers.map(esc).join(',')];
    for(const row of arr){ lines.push(headers.map(h=>esc(row[h])).join(',')); }
    return lines.join('
');
  }

  // --- Rendering ------------------------------------------------------------
  function renderFeatures(){
    const wrap = document.getElementById('featureTableWrap');
    const count = document.getElementById('featCount');
    if(!wrap||!count) return;
    count.textContent = state.featureMasterList.length.toString();

    if(!state.featureMasterList.length){ wrap.innerHTML = '<div class="muted">No features loaded.</div>'; return; }

    const cols = [
      'Generic Feature','SG (Y/N)','PB (Y/N)','BS (Y/N)',
      'MOQ (units)','Lead Time (weeks)','Compatible Price Points','Exclusivity Notes','Green Compliant (Y/N)','Requires Pre-Approval (Y/N)'
    ];

    let html = '<div style="overflow:auto;max-height:58vh">';
    html += '<table><thead><tr>' + cols.map(c=>`<th>${c}</th>`).join('') + '</tr></thead><tbody>';
    for(const f of state.featureMasterList){
      html += '<tr>' + cols.map(c=>`<td>${(f[c]||'').toString()}</td>`).join('') + '</tr>';
    }
    html += '</tbody></table></div>';
    wrap.innerHTML = html;
  }

  function renderDenoms(){
    const mount = document.getElementById('denoms');
    if(!mount) return;
    if(!state.denominations.length){
      mount.innerHTML = '<div class="muted">No denominations yet. (Add UI in next step.)</div>';
      return;
    }
  }

  function renderProgress(){
    const fill = document.getElementById('progressFill');
    const txt = document.getElementById('progressText');
    if(!fill||!txt) return;
    const pct = 0; // placeholder until allocations exist
    fill.style.width = pct + '%';
    txt.textContent = `Allocated: ${pct}% of sales target`;
  }

  function render(){
    renderFeatures();
    renderDenoms();
    renderProgress();
  }

  // --- IO: Save/Load Plan (JSON) -------------------------------------------
  async function savePlan(){
    try{
      const handle = await window.showSaveFilePicker({suggestedName:`fy-plan-${state.fiscalYear}.json`,types:[{description:'Plan JSON',accept:{'application/json':['.json']}}]});
      const writable = await handle.createWritable();
      await writable.write(JSON.stringify(state,null,2));
      await writable.close();
      logAudit('Saved plan to disk.');
    }catch(err){ if(err&&err.name!=="AbortError") console.error(err); }
  }

  async function loadPlan(){
    try{
      const [handle] = await window.showOpenFilePicker({types:[{description:'Plan JSON',accept:{'application/json':['.json']}}]});
      const file = await handle.getFile();
      const json = JSON.parse(await file.text());
      Object.assign(state, json);
      document.getElementById('fy').value = state.fiscalYear;
      const salesEl = document.getElementById('sales');
      if(salesEl) salesEl.value = formatNumber(state.totalSalesTarget||0);
      render();
      logAudit('Loaded plan from disk.');
    }catch(err){ if(err&&err.name!=="AbortError") console.error(err); }
  }

  // --- Number formatting for Total Sales Target ----------------------------
  function formatNumber(num){
    const n = Number(num);
    if(!isFinite(n) || n<=0) return '';
    return n.toLocaleString('en-US');
  }

  function parseNumber(str){
    return Number(String(str||'').replace(/,/g, '')) || 0;
  }

  // --- Bootstrap ------------------------------------------------------------
  function init(){
    // Seed header fields
    const fyEl = document.getElementById('fy');
    const salesEl = document.getElementById('sales');
    if(fyEl) fyEl.value = state.fiscalYear;

    if(salesEl){
      const formatAndSet = (el) => {
        const caret = el.selectionStart ?? el.value.length;
        const beforeLen = el.value.length;
        const raw = parseNumber(el.value);
        el.value = formatNumber(raw);
        const afterLen = el.value.length;
        const offset = afterLen - beforeLen;
        try { el.setSelectionRange(Math.max(0, caret + offset), Math.max(0, caret + offset)); } catch(_){/* noop */}
      };

      // Live format while typing
      salesEl.addEventListener('input', (e)=>{ formatAndSet(e.target); });
      // Commit on blur
      salesEl.addEventListener('blur', (e)=>{
        formatAndSet(e.target);
        state.totalSalesTarget = parseNumber(e.target.value);
        logAudit(`Updated Total Sales Target to $${formatNumber(state.totalSalesTarget)}`);
        renderProgress();
      });
      // Enter commits
      salesEl.addEventListener('keydown', (e)=>{ if(e.key==='Enter'){ e.preventDefault(); salesEl.blur(); }});

      // Format initial placeholder value if user pre-fills
      if(salesEl.value){ salesEl.value = formatNumber(parseNumber(salesEl.value)); }
    }

    document.getElementById('btnSave')?.addEventListener('click', savePlan);
    document.getElementById('btnLoad')?.addEventListener('click', loadPlan);
    document.getElementById('btnExportFeatures')?.addEventListener('click', ()=>{
      const csv = toCSV(state.featureMasterList);
      const blob = new Blob([csv],{type:'text/csv'});
      const url = URL.createObjectURL(blob);
      const a = Object.assign(document.createElement('a'),{href:url,download:'vendor_features.csv'});
      document.body.appendChild(a); a.click(); a.remove();
      URL.revokeObjectURL(url);
      logAudit('Exported feature database as CSV.');
    });

    // Load embedded CSV
    const csvEl = document.getElementById('features-csv');
    if(csvEl){
      try{
        state.featureMasterList = parseCSV(csvEl.textContent.trim());
        logAudit(`Loaded ${state.featureMasterList.length} vendor features from embedded CSV.`);
      }catch(err){ console.error('Failed to parse embedded CSV', err); }
    }

    render();
  }

  document.addEventListener('DOMContentLoaded', init);
})();
</script>
