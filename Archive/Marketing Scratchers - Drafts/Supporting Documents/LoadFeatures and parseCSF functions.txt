(function() {
    'use strict';

    const state = {
        // ... (previous state properties)
        featureMasterList: []
    };

    /**
     * Initializes the application.
     */
    function init() {
        console.log("Application Initialized.");
        loadFeatures();
    }

    /**
     * Fetches and parses the master feature CSV file.
     */
    async function loadFeatures() {
        try {
            const response = await fetch('features.csv');
            if (!response.ok) {
                throw new Error(`HTTP error! status: ${response.status}`);
            }
            const csvText = await response.text();
            state.featureMasterList = parseCSV(csvText);
            console.log("✅ Vendor features loaded successfully.");
            // Log a sample to verify
            console.log("Sample feature:", state.featureMasterList[0]); 
        } catch (error) {
            console.error("❌ Failed to load features:", error);
            // We could render an error message to the user here.
        }
        render(); // Render the UI now that we have data
    }
    
    /**
     * A simple CSV parser to convert CSV text to an array of objects.
     * @param {string} csvText The raw CSV string data.
     * @returns {Array<Object>}
     */
    function parseCSV(csvText) {
        const lines = csvText.trim().split('\n');
        const headers = lines.shift().split(',').map(h => h.trim().replace(/"/g, ''));
        return lines.map(line => {
            const values = line.split(',').map(v => v.trim().replace(/"/g, ''));
            return headers.reduce((obj, header, index) => {
                obj[header] = values[index];
                return obj;
            }, {});
        });
    }

    /**
     * Renders the entire UI based on the current state.
     */
    function render() {
        console.log("Rendering UI with loaded data...");
        // In the next steps, this function will build the HTML tables and lists.
    }

    document.addEventListener('DOMContentLoaded', init);

})();
